    function generatePOIs(centerLat, centerLon, count = 20) {
        if (layoutMode === 'random') return generateRandomPOIs(centerLat, centerLon, count);
        return generateGridPOIs(centerLat, centerLon, count);
    }

    function generateGridPOIs(centerLat, centerLon, count) {
        const pois = [];
        const kmInLat = 0.009;
        const kmInLon = 0.009 / Math.cos(centerLat * Math.PI / 180);
        const gridSize = Math.ceil(Math.sqrt(count));
        const totalSize = gridSize - 1;
        
        const startLat = centerLat - (totalSize / 2) * kmInLat;
        const startLon = centerLon - (totalSize / 2) * kmInLon;
        
        let built = 0;
        for (let row = 0; row < gridSize; row++) {
            for (let col = 0; col < gridSize; col++) {
                if (built >= count) break;
                
                const lat = startLat + (row * kmInLat);
                const lon = startLon + (col * kmInLon);
                const buildingName = `${buildingNames[selectedBuildingType]} ${currentCityName} ${built + 1}`;
                
                pois.push({ 
                    name: buildingName, 
                    lat: lat.toFixed(6), 
                    lon: lon.toFixed(6), 
                    city: currentCityName,
                    type: selectedBuildingType
                });
                built++;
            }
        }
        return pois;
    }

    function generateRandomPOIs(centerLat, centerLon, count) {
        const pois = [];
        const maxRadius = Math.ceil(Math.sqrt(count) * 0.5);
        
        for (let i = 0; i < count; i++) {
            let attempts = 0;
            let validPosition = false;
            let lat, lon;
            
            while (!validPosition && attempts < 100) {
                const angle = Math.random() * 2 * Math.PI;
                const distance = 1 + (Math.random() * (maxRadius - 1));
                
                const deltaLat = (distance / 111) * Math.cos(angle);
                const deltaLon = (distance / (111 * Math.cos(centerLat * Math.PI / 180))) * Math.sin(angle);
                
                lat = centerLat + deltaLat;
                lon = centerLon + deltaLon;
                
                validPosition = true;
                for (const poi of pois) {
                    const dist = calculateDistance(lat, lon, parseFloat(poi.lat), parseFloat(poi.lon));
                    if (dist < 1.0) {
                        validPosition = false;
                        break;
                    }
                }
                attempts++;
            }
            
            const buildingName = `${buildingNames[selectedBuildingType]} ${currentCityName} ${i + 1}`;
            pois.push({ 
                name: buildingName, 
                lat: lat.toFixed(6), 
                lon: lon.toFixed(6), 
                city: currentCityName,
                type: selectedBuildingType
            });
        }
        return pois;
    }

    function calculateDistance(lat1, lon1, lat2, lon2) {
        const R = 6371;
        const dLat = (lat2 - lat1) * Math.PI / 180;
        const dLon = (lon2 - lon1) * Math.PI / 180;
        const a = 
            Math.sin(dLat/2) * Math.sin(dLat/2) +
            Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * 
            Math.sin(dLon/2) * Math.sin(dLon/2);
        const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
        return R * c;
    }

    function createDraggableMarker(lat, lng, name, index) {

            html: `<div style="background-color: #e74c3c; border: 3px solid #c0392b; width: 20px; height: 20px; border-radius: 50%; cursor: move;"></div>`,
            iconSize: [20, 20],
            iconAnchor: [10, 10]
        });

        const marker = L.marker([lat, lng], {
            icon: editIcon,
            draggable: isEditMode
        });

        marker.bindPopup(`
            <strong>${name}</strong><br>
            Position: ${lat}, ${lng}<br>
            <button onclick="window.updatePoiPosition(${index}, ${lat}, ${lng})" class="btn btn-xs btn-success">Position speichern</button>
            <button onclick="window.removePoi(${index})" class="btn btn-xs btn-danger">Entfernen</button>
        `);

        marker.on('dragend', function(e) {
            const newLat = e.target.getLatLng().lat;
            const newLng = e.target.getLatLng().lng;
            window.updatePoiPosition(index, newLat, newLng);
            
            marker.getPopup().setContent(`
                <strong>${name}</strong><br>
                Position: ${newLat.toFixed(6)}, ${newLng.toFixed(6)}<br>
                <button onclick="window.updatePoiPosition(${index}, ${newLat}, ${newLng})" class="btn btn-xs btn-success">Position speichern</button>
                <button onclick="window.removePoi(${index})" class="btn btn-xs btn-danger">Entfernen</button>
            `);
        });

        return marker;
    }

    function updatePoiPosition(index, lat, lng) {
        if (index >= 0 && index < currentPois.length) {
            currentPois[index].lat = lat.toFixed(6);
            currentPois[index].lon = lng.toFixed(6);
            
            if (map) map.removeLayer(currentMarkers[index]);
            currentMarkers[index] = createDraggableMarker(lat, lng, currentPois[index].name, index);
            currentMarkers[index].addTo(map);
            
            if (window.bepolInfo) window.bepolInfo.textContent = `‚úÖ Position von ${currentPois[index].name} aktualisiert`;
        }
    }

    function removePoi(index) {
        if (index >= 0 && index < currentPois.length) {
            const name = currentPois[index].name;
            
            if (map) map.removeLayer(currentMarkers[index]);
            currentPois.splice(index, 1);
            currentMarkers.splice(index, 1);
            
            currentMarkers.forEach((marker, newIndex) => {
                if (marker && marker.getPopup()) {
                    const poi = currentPois[newIndex];
                    marker.getPopup().setContent(`
                        <strong>${poi.name}</strong><br>
                        Position: ${poi.lat}, ${poi.lon}<br>
                        <button onclick="window.updatePoiPosition(${newIndex}, ${parseFloat(poi.lat)}, ${parseFloat(poi.lon)})" class="btn btn-xs btn-success">Position speichern</button>
                        <button onclick="window.removePoi(${newIndex})" class="btn btn-xs btn-danger">Entfernen</button>
                    `);
                }
            });

            if (currentPois.length === 0 && window.autoBuildButton) {
                window.autoBuildButton.disabled = true;
            }
        }

    async function configureBuildingPersonnel(buildingId, buildingType) {
        return new Promise((resolve) => {
            const targetPersonnel = personnelSettings[buildingType];
            
            if (!targetPersonnel) {
                resolve(false);
                return;
            }

            // Personal-Sollst√§rke setzen
            setTimeout(() => {
                const csrfToken = document.querySelector('meta[name="csrf-token"]')?.content;
                if (!csrfToken) {
                    resolve(false);
                    return;
                }

                const formData = new URLSearchParams();
                formData.append('_method', 'put');
                formData.append('authenticity_token', csrfToken);
                formData.append('building[personal_count_target]', targetPersonnel.toString());

                fetch(`/buildings/${buildingId}?personal_count_target_only=1`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/x-www-form-urlencoded',
                    },
                    body: formData.toString()
                })
                .then(response => {
                    if (response.ok) {
                        // Automatisches Werben aktivieren
                        return fetch(`/buildings/${buildingId}/hire_do/automatic`);
                    }
                    return Promise.reject('Fehler beim Setzen der Personalst√§rke');
                })
                .then(response => {
                    if (response.ok) {
                        console.log(`‚úÖ Geb√§ude ${buildingId} konfiguriert: ${targetPersonnel} Personal + automatisches Werben`);
                        resolve(true);
                    } else {
                        resolve(false);
                    }
                })
                .catch(error => {
                    console.log('‚ùå Fehler bei Personal-Konfiguration:', error);
                    resolve(false);
                });
            }, 2000);
        });
    }

    async function buildAtLocation(lat, lon, name, buildingType = '11', isLast = false) {
        return new Promise((resolve) => {
            const checkMenu = setInterval(() => {
                const nameInput = document.getElementById('building_name');
                const typeSelect = document.getElementById('building_building_type');
                if (nameInput && typeSelect) {
                    clearInterval(checkMenu);

                    setBuildAnotherCheckbox(!isLast);

                    nameInput.value = name;
                    typeSelect.value = buildingType;
                    document.getElementById('building_latitude').value = lat;
                    document.getElementById('building_longitude').value = lon;
                    if (selectedLeitstelle) setLeitstelle(selectedLeitstelle);

                    setTimeout(() => {
                        const buildBtn = document.querySelector(`#detail_${buildingType} .build_with_credits_step`);
                        if (buildBtn) buildBtn.click();

                        // Warte auf Bau-Abschluss
                        setTimeout(async () => {
                            if (document.querySelector('.alert-success')) {
                                // Erfolgreich gebaut - jetzt Building-ID ermitteln und Personal konfigurieren
                                const buildingId = await getLatestBuildingId();
                                if (buildingId) {
                                    await configureBuildingPersonnel(buildingId, parseInt(buildingType));
                                }
                                resolve(true);
                            } else {
                                if (isLast) {
                                    const backButton = document.getElementById('building_back_button');
                                    if (backButton) backButton.click();
                                }
                                resolve(false);
                            }
                        }, 4000);
                    }, 1200);
                }
            }, 200);
            setTimeout(() => clearInterval(checkMenu), 4000);
        });
    }

    async function getLatestBuildingId() {
        return new Promise((resolve) => {
            // Kurz warten bis das Geb√§ude in der API verf√ºgbar ist
            setTimeout(() => {
                GM_xmlhttpRequest({
                    method: 'GET',
                    url: 'https://www.leitstellenspiel.de/api/buildings',
                    onload: function(response) {
                        try {
                            const buildings = JSON.parse(response.responseText);
                            if (buildings && buildings.length > 0) {
                                // Neuestes Geb√§ude (h√∂chste ID) finden
                                const latestBuilding = buildings.reduce((latest, current) => 
                                    current.id > latest.id ? current : latest
                                );
                                resolve(latestBuilding.id);
                            } else {
                                resolve(null);
                            }
                        } catch {
                            resolve(null);
                        }
                    },
                    onerror: () => resolve(null)
                });
            }, 3000);
        });
    }

    async function startAutoBuild(pois) {
        if (isBuilding) return;
        isBuilding = true;
        
        if (window.bepolInfo) {
            window.bepolInfo.innerHTML = "üèóÔ∏è Starte automatischen Bau mit Personal-Konfiguration...<br>" +
                                       "üìã Personal-Sollst√§rken und automatisches Werben werden gesetzt";
        }

        if (!openBuildMenu()) {
            if (window.bepolInfo) window.bepolInfo.textContent = "‚ùå Bau-Men√º konnte nicht ge√∂ffnet werden!";
            isBuilding = false;
            return;
        }

        await new Promise(r => setTimeout(r, 3000));

        let successCount = 0;
        let personalConfigCount = 0;

        for (let i = 0; i < pois.length && isBuilding; i++) {
            const statusText = `üèóÔ∏è Baue ${pois[i].name} (${i + 1}/${pois.length})...`;
            if (window.bepolInfo) window.bepolInfo.textContent = statusText;
            
            const isLastBuilding = (i === pois.length - 1);
            const buildSuccess = await buildAtLocation(pois[i].lat, pois[i].lon, pois[i].name, selectedBuildingType, isLastBuilding);

            if (buildSuccess) {
                successCount++;
                personalConfigCount++;
            }

            // Status-Update
            if (window.bepolInfo) {
                window.bepolInfo.innerHTML = 
                    `‚úÖ ${successCount}/${pois.length} Geb√§ude gebaut<br>` +
                    `üìã ${personalConfigCount} Personal-Konfigurationen durchgef√ºhrt<br>` +
                    `‚è≥ N√§chstes: ${pois[i].name}`;
            }

            if (!isLastBuilding) {
                await new Promise(r => setTimeout(r, 7000));

                const waitForForm = setInterval(() => {
                    const nameInput = document.getElementById('building_name');
                    if (nameInput && nameInput.value === '') {
                        clearInterval(waitForForm);
                    }
                }, 100);

                await new Promise(r => setTimeout(r, 1000));
            }
        }

        if (window.bepolInfo) {
            window.bepolInfo.innerHTML = 
                `‚úÖ Automatischer Bau abgeschlossen!<br>` +
                `üèóÔ∏è ${successCount} Geb√§ude gebaut<br>` +
                `üìã ${personalConfigCount} Personal-Konfigurationen durchgef√ºhrt`;
        }
        
        isBuilding = false;
        if (window.autoBuildButton) {
            window.autoBuildButton.textContent = "üèóÔ∏è Automatisch bauen + Personal";
            window.autoBuildButton.classList.remove('building');
        }
    }

    function stopAutoBuild() {
        isBuilding = false;
        setBuildAnotherCheckbox(false);
        const backButton = document.getElementById('building_back_button');
        if (backButton) backButton.click();

        if (window.bepolInfo) window.bepolInfo.textContent = "‚èπÔ∏è Automatischer Bau gestoppt.";
        if (window.autoBuildButton) {
            window.autoBuildButton.textContent = "üèóÔ∏è Automatisch bauen + Personal";
            window.autoBuildButton.classList.remove('building');
        }
    }

                    const success = await fetchUserBuildings();
                    const sel = document.getElementById('leitstelleSelect');
                    sel.innerHTML = '<option value="">Keine Leitstelle</option>';

                        list.forEach(l => {

                document.getElementById('searchButton').onclick = () => {
                    const city = document.getElementById('cityInput').value.trim();
                    const count = parseInt(document.getElementById('poiCount').value) || 10;
                    
                    if (!city) {
                        window.bepolInfo.textContent = "‚ùå Bitte Stadtnamen eingeben!";
                        return;
                    }
                    
                    if (count < 1 || count > 50) {
                        window.bepolInfo.textContent = "‚ùå Anzahl muss zwischen 1 und 50 liegen!";
                        return;
                    }

                    window.bepolInfo.textContent = 'üîé Suche Stadt...';
                    fetch(`https://nominatim.openstreetmap.org/search?city=${encodeURIComponent(city)}&format=json&limit=1`)
                        .then(r => r.json())
                        .then(data => {
                            if (!data.length) throw new Error('Stadt nicht gefunden');
                            const lat = parseFloat(data[0].lat);
                            const lon = parseFloat(data[0].lon);
                            currentCityName = data[0].display_name.split(',')[0];
                            clearMarkers();
                            currentPois = generatePOIs(lat, lon, count);
                            
                            currentPois.forEach((poi, index) => {
                                const marker = createDraggableMarker(
                                    parseFloat(poi.lat), 
                                    parseFloat(poi.lon), 
                                    poi.name, 
                                    index
                                );
                                marker.addTo(map);
                                currentMarkers.push(marker);
                            });
                            
                            map.setView([lat, lon], 12);
                            const personnelSetting = personnelSettings[selectedBuildingType] || 'Automatisch';
                            window.bepolInfo.innerHTML = 
                                `‚úÖ ${currentPois.length} ${buildingNames[selectedBuildingType]} in ${currentCityName}<br>` +
                                `üìè 1km Mindestabstand zwischen Wachen<br>` +
                                `üë• Personal-Soll: ${personnelSetting}`;
                            window.autoBuildButton.disabled = false;
                        })
                        .catch(err => { 
                            window.bepolInfo.textContent = '‚ùå ' + err.message; 
                        });
                };

                // Dragbar f√ºr UI
                ui.onmousedown = function(e) {
                    let shiftX = e.clientX - ui.getBoundingClientRect().left;
                    let shiftY = e.clientY - ui.getBoundingClientRect().top;
                    function moveAt(pageX, pageY) {
                        ui.style.left = pageX - shiftX + 'px';
                        ui.style.top = pageY - shiftY + 'px';
                    }
                    function onMouseMove(e) { moveAt(e.pageX, e.pageY); }
                    document.addEventListener('mousemove', onMouseMove);
                    ui.onmouseup = function() {
                        document.removeEventListener('mousemove', onMouseMove);
                        ui.onmouseup = null;
                    };
                };
                ui.ondragstart = () => false;
            }
        }, 1000);
